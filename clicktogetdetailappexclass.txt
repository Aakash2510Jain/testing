	public class clicktogetdetail {
	public List<Account> accounts{get;set;}
	public string accid{get;set;}
	public List<contact> contacts{get;set;}
	public List<opportunity> opportunities{get;set;}
	public List<case> cases{get;set;}
	public string conliststring{get;set;}
	@RemoteAction
	public static List<contact> getcontacts(string id)
	{        
	List<contact> contacts= [SELECT id,firstname,lastname from contact WHERE accountid=:id];
	system.debug('####'+contacts);
	return contacts;
	}

	@RemoteAction
	public static List<opportunity> getopportunities(string id)
	{        
	List<opportunity> opportunities= [SELECT id,Name,Account.Name ,CloseDate from opportunity WHERE accountid=:id];
	system.debug('####'+opportunities);
	return opportunities;
	}
	@RemoteAction
	public static List<case> getcases(string id)
	{        
	List<case> cases= [SELECT id,CaseNumber,CreatedDate,Priority from case WHERE accountid=:id];
	system.debug('####'+cases);
	return cases;
	}

	}
